        theme: ThemeData(

MaterialApp implements the Material Design guidelines, providing a consistent look and feel for your application. This includes predefined layouts, color schemes, typography, and components such as buttons, cards, and dialogs.

Ensure that the "screens" subfolder is located in the lib folder. An assets folder must also be created in your project root with an images subfolder. We are using 6 random images with simplified names (example: dog.jpg) placed in the assets->images subfolder. These images can be any images you find online or already stored in your computer.

Also note that the pubspec.yaml file must have this entry to link the images to your project:

flutter:
  uses-material-design: true

  assets:
    - assets/images/  


import 'package:flutter/material.dart'; //always imported on all screens

/*
simple class generated by 'stl' in VSCode.  Page/class is called MyHomePage in PascalCase
*/

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key});

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}


//Below is the companion class that inherits from the State class
class _MyHomePageState extends State<MyHomePage> {
  @override
  Widget build(BuildContext context) {
  
    /* Here we declare two string lists: one returning relative paths for our images,
    the other returning labels to be placed on top of them.  We will place these lists
    dynamically to avoid bloated code.
    */
    
    
    List<String> imageList = [
      'assets/images/bark.jpg',
      'assets/images/cat.jpg',
      'assets/images/city.png',
      'assets/images/hippo.jpg',
      'assets/images/iceland.jpg',
      'assets/images/jellyfish.jpg'
    ];

    List<String> imageTitle = [
      'tree',
      'cat',
      'city',
      'hippo',
      'iceland',
      'jellyfish'
    ];
    

    //we find the viewport width and height with mediaquery and context
    double Width = MediaQuery.of(context).size.width;
    double Height = MediaQuery.of(context).size.height;
    
    
    /* 
    
    Below we define a list widget returning ui widgets.  This is the structure
    of a single grid item using Container() widget and its decorative properties.
    The List.generate argument of 6 must correspond to the numbers of items a list          possesses that you are passing in (defined above).  The labels are customized
    with the Text() widget using color, font-size, weight, and shadowing.
    */
    
    
    List<Widget> gridStuff = List.generate(6, (index) {
      return Container(
        width: Width,
        height: Height,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(4),
          image: DecorationImage(
            image: AssetImage(imageList[index]),
            fit: BoxFit.cover,
          ),
        ),
        child: Center(
          child: Text(
            imageTitle[index],
            style: TextStyle(
              color: Colors.white,
              fontSize: 42,
              fontWeight: FontWeight.bold,
              shadows: [
                Shadow(
                  blurRadius: 5,
                  color: Colors.black.withOpacity(0.5),
                  offset: Offset(5, 5),
                ),
              ],
            ),
          ),
        ),
      );
    });


    /*
    Our Scaffold is the blueprint of our screen.  Here we are able to use the appBar, 
    body, drawer, & bottomNavigationBar properties.  
    
    In the appBar we have a simple Row() widget with Text() and ElevatedButton()            widgets.  These are spaced apart using the Row()'s mainAxisAlignment property
    and a SizedBoxed() widget where we can manipulate height and width.  
    */
    

    return Scaffold(
      appBar: AppBar(
        backgroundColor: Colors.red,
        title: const Row(
          mainAxisAlignment: MainAxisAlignment.spaceBetween,
          children: [
            Text(
              'Home Page',
              style: TextStyle(fontWeight: FontWeight.w900),
            ),
            SizedBox(
              width: 120,
            ),
            ElevatedButton(
              onPressed: null,
              child: Icon(Icons.notification_add, color: Colors.white),
            ),
            Icon(Icons.settings),
          ],
        ),
      ),
      drawer: Drawer(),
      
      /*
      Here in the body property we return the GridView.builder with allows us to build
      the dynamic grid.  the crossAxisCount gives us the ability to dictate how many          grid rows span across the screen and mainAxisExtent will determine the height
      of those grid items.
      
      We then use ItemBuilder to build out the UI portion of this dynamically, passing
      in the gridStuff blueprint widget list we defined above.  Note that it 
      takes in an index of the gridStuff.length, which is simply 6.
      */
      
      
      body: Container(
        decoration: const BoxDecoration(
          color: Colors.red,
        ),
        child: GridView.builder(
            gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
              crossAxisCount: 2,
              mainAxisExtent: 240,
            ),
            itemCount: gridStuff.length,
            itemBuilder: (BuildContext context, int index) {
              return Column(
                children: [
                  Container(height: 240, child: gridStuff[index]),
                ],
              );
            }),
      ),
      
      /* 
      The bottomNavigationBar takes BottomNavigationBarItem() widget which 
      basically just Icon() widgets and labels (simple strings).  The
      BottomNavigationBar has a currentIndex & onTap() property which we normally
      use for click actions (buttons for navigation), but this is not implemented
      here as we are focusing on strictly layout.
      
      */
      
      
      
      
      bottomNavigationBar:
          BottomNavigationBar(showUnselectedLabels: true, items: const [
        BottomNavigationBarItem(
            backgroundColor: Colors.red, icon: Icon(Icons.home), label: 'Home'),
        BottomNavigationBarItem(
            backgroundColor: Colors.red,
            icon: Icon(Icons.search),
            label: 'Search'),
        BottomNavigationBarItem(
            backgroundColor: Colors.red,
            icon: Icon(Icons.settings),
            label: 'Settings'),
        BottomNavigationBarItem(
            backgroundColor: Colors.red,
            icon: Icon(Icons.person),
            label: 'Profile'),
      ]),
    );
  }
}
